class Solution {
    public boolean isPalindrome(int x) {
        if(x<0)
        {
             return false;
        }

          int n=x;
          int ans=0;

          while(n>0)
          {
            int d=n%10;
            ans=ans*10+d;
            n=n/10;

          } 
          if(ans==x)
          {
            return true;
          }
          else
          {
            return false;
          }
      

    }
}
